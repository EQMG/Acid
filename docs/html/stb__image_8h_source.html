5 //
6 // Limitations:
7 // - no jpeg progressive support
8 // - non-HDR formats support 8-bit samples only (jpeg, png)
9 // - no delayed line count (jpeg) -- IJG doesn't support either
10 // - no 1-bit BMP
11 // - GIF always returns *comp=4
12 //
13 // Basic usage (see HDR discussion below):
14 // int x,y,n;
15 // unsigned char *data = stbi_load(filename, &x, &y, &n, 0);
16 // // ... process data if not NULL ...
17 // // ... x = width, y = height, n = # 8-bit components per pixel ...
18 // // ... replace '0' with '1'..'4' to force that many components per pixel
19 // // ... but 'n' will always be the number that it would have been if you said 0
20 // stbi_image_free(data)
21 //
22 // Standard parameters:
23 // int *x -- outputs image width in pixels
24 // int *y -- outputs image height in pixels
25 // int *comp -- outputs # of image components in image file
26 // int req_comp -- if non-zero, # of image components requested in result
27 //
28 // The return value from an image loader is an 'unsigned char *' which points
29 // to the pixel data. The pixel data consists of *y scanlines of *x pixels,
30 // with each pixel consisting of N interleaved 8-bit components; the first
31 // pixel pointed to is top-left-most in the image. There is no padding between
32 // image scanlines or between pixels, regardless of format. The number of
33 // components N is 'req_comp' if req_comp is non-zero, or *comp otherwise.
34 // If req_comp is non-zero, *comp has the number of components that _would_
35 // have been output otherwise. E.g. if you set req_comp to 4, you will always
36 // get RGBA output, but you can check *comp to easily see if it's opaque.
37 //
38 // An output image with N components has the following components interleaved
39 // in this order in each pixel:
40 //
41 // N=#comp components
42 // 1 grey
43 // 2 grey, alpha
44 // 3 red, green, blue
45 // 4 red, green, blue, alpha
46 //
47 // If image loading fails for any reason, the return value will be NULL,
48 // and *x, *y, *comp will be unchanged. The function stbi_failure_reason()
49 // can be queried for an extremely brief, end-user unfriendly explanation
50 // of why the load failed. Define STBI_NO_FAILURE_STRINGS to avoid
51 // compiling these strings at all, and STBI_FAILURE_USERMSG to get slightly
52 // more user-friendly ones.
53 //
54 // Paletted PNG, BMP, GIF, and PIC images are automatically depalettized.
55 //
56 // ===========================================================================
57 //
58 // iPhone PNG support:
59 //
60 // By default we convert iphone-formatted PNGs back to RGB; nominally they
61 // would silently load as BGR, except the existing code should have just
62 // failed on such iPhone PNGs. But you can disable this conversion by
63 // by calling stbi_convert_iphone_png_to_rgb(0), in which case
64 // you will always just get the native iphone "format" through.
65 //
66 // Call stbi_set_unpremultiply_on_load(1) as well to force a divide per
67 // pixel to remove any premultiplied alpha *only* if the image file explicitly
68 // says there's premultiplied data (currently only happens in iPhone images,
69 // and only if iPhone convert-to-rgb processing is on).
70 //
71 // ===========================================================================
72 //
73 // HDR image support (disable by defining STBI_NO_HDR)
74 //
75 // stb_image now supports loading HDR images in general, and currently
76 // the Radiance .HDR file format, although the support is provided
77 // generically. You can still load any file through the existing interface;
78 // if you attempt to load an HDR file, it will be automatically remapped to
79 // LDR, assuming gamma 2.2 and an arbitrary scale factor defaulting to 1;
80 // both of these constants can be reconfigured through this interface:
81 //
82 // stbi_hdr_to_ldr_gamma(2.2f);
83 // stbi_hdr_to_ldr_scale(1.0f);
84 //
85 // (note, do not use _inverse_ constants; stbi_image will invert them
86 // appropriately).
87 //
88 // Additionally, there is a new, parallel interface for loading files as
89 // (linear) floats to preserve the full dynamic range:
90 //
91 // float *data = stbi_loadf(filename, &x, &y, &n, 0);
92 //
93 // If you load LDR images through this interface, those images will
94 // be promoted to floating point values, run through the inverse of
95 // constants corresponding to the above:
96 //
97 // stbi_ldr_to_hdr_scale(1.0f);
98 // stbi_ldr_to_hdr_gamma(2.2f);
99 //
100 // Finally, given a filename (or an open file or memory block--see header
101 // file for details) containing image data, you can query for the "most
102 // appropriate" interface to use (that is, whether the image is HDR or
103 // not), using:
104 //
105 // stbi_is_hdr(char *filename);
106 //
107 // ===========================================================================
108 //
109 // I/O callbacks
110 //
111 // I/O callbacks allow you to read from arbitrary sources, like packaged
112 // files or some other source. Data read from callbacks are processed
113 // through a small internal buffer (currently 128 bytes) to try to reduce
114 // overhead.
115 //
116 // The three functions you must define are "read" (reads some bytes of data),
117 // "skip" (skips some bytes of data), "eof" (reports if the stream is at the end).
